leetcode 1187. 使数组严格递增

发表于 2019-09-09 | 分类于 leetcode题目 | 评论数：

题目：

给你两个整数数组 arr1 和 arr2，返回使 arr1 严格递增所需要的最小「操作」数（可能为 0）。

每一步「操作」中，你可以分别从 arr1 和 arr2 中各选出一个索引，分别为 i 和 j，0 <= i < arr1.length 和 0 <= j < arr2.length，然后进行赋值运算 arr1[i] = arr2[j]。

如果无法让 arr1 严格递增，请返回 -1。

示例：

示例 1：

输入：arr1 = [1,5,3,6,7], arr2 = [1,3,2,4]
输出：1
解释：用 2 来替换 5，之后 arr1 = [1, 2, 3, 6, 7]。

示例 2：

输入：arr1 = [1,5,3,6,7], arr2 = [4,3,1]
输出：2
解释：用 3 来替换 5，然后用 4 来替换 3，得到 arr1 = [1, 3, 4, 6, 7]。

示例 3：

输入：arr1 = [1,5,3,6,7], arr2 = [1,6,3,3]
输出：-1
解释：无法使 arr1 严格递增。

提示：

  • 1 <= arr1.length, arr2.length <= 2000
  • 0 <= arr1[i], arr2[i] <= 10^9

分析：

将arr2去重排序

动态规划

定义一个函数dy，用来计算arr2中大于传入参数z的第一个值

dp[i][j],i为第i个数，j为修改j次

在j为0的时候，根据之前的数是否是递增关系来判断是否将dp[i][0]置为arr1[i]

dp[0][1] = arr2中最小的值

dp[i][j] = dy(arr2，dp[i][j-1]) || arr1[i] (需要判断arr1[i]是否大于vvi[i-1][j])

最后从dp[dp.size()-1][0]开始遍历一遍，返回第一个非-1值时的j

代码：

class Solution {
public:
    int dy(vector<int> si, int z){
        int l = 0; 
        int r = si.size()-1;
        int mid;
        while(l <= r){
            mid = l + (r - l) / 2;
            if(z < si[mid]){
                r = mid - 1;
            }else{
                l = mid + 1;
            }

        }
        // cout << "<" << z << ">";
        return l >= si.size() ? -1 : si[l];
    }
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        vector<vector<int>> vvi(arr1.size(), vector<int>(arr1.size()+1, -1));
        set<int> si(arr2.begin(), arr2.end());
        arr2.assign(si.begin(), si.end());

        for(int i = 0; i < arr1.size(); i++){
            for(int j = 0; j <= i+1; j++){
                if(j == 0 && (i-1 < 0 || (vvi[i-1][0] != -1 && arr1[i] > arr1[i-1]))){
                    vvi[i][0] = arr1[i];
                }
                else{
                    // if(i-1 >= 0 && j-1 >= 0 && vvi[i-1][j-1] == -1){
                    //     continue;
                    // }
                    int d = INT_MAX; // 根据j-1获得，这次改变
                    int d1 = INT_MAX; // 根据j获得，这次不变
                    if(i-1 >= 0 && j-1 >= 0 && vvi[i-1][j-1] != -1){
                        d = dy(arr2, vvi[i-1][j-1]);
                        d = d > -1 ? d : INT_MAX;
                    }
                    if(i - 1 >= 0 && vvi[i-1][j] != -1 && vvi[i-1][j] < arr1[i]){
                        d1 = arr1[i];
                    }

                    if(d == INT_MAX && d1 == INT_MAX)
                        vvi[i][j] = -1;
                    else
                        vvi[i][j] = min(d1, d);

                    if(i == 0 && j == 1){ // 在第0位1次修改的时候，也就是第一个数字改为arr2中最小的那个值
                        vvi[0][1] = arr2[0];
                    }
                }
                //cout << vvi[i][j] << "\t";
            }
            // cout << "\n";
        }
        for(int i = 0; i < arr1.size()+1; i++){
            if(vvi[arr1.size()-1][i] != -1){
                return i;
            }
        }
        return -1;
    }
};
